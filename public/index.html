import express from "express"; import cors from "cors"; import fetch from "node-fetch"; const app = express(); app.use(cors()); app.use(express.json()); app.post("/chat", async (req, res) => { try { const userMessage = req.body.message; const response = await fetch("https://api.openai.com/v1/responses", { method: "POST", headers: { "Authorization": `Bearer ${process.env.OPENAI_API_KEY}`, "Content-Type": "application/json" }, body: JSON.stringify({ model: "gpt-4.1-mini", input: `Eres el asistente de Chambatina. Responde claro y enfocado en envíos a Cuba.\nCliente: ${userMessage}` }) }); const data = await response.json(); res.json({ reply: data.output[0].content[0].text }); } catch (error) { console.error(error); res.json({ reply: "Error en el servidor" }); } }); app.listen(3000, () => { console.log("Servidor corriendo"); });
